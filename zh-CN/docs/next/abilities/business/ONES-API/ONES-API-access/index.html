): Promise<PluginResponse> {

    path: `/users/me`,

    headers: {
      'Ones-User-Id': [userUUID],
      'Ones-Auth-Token': [userToken],
    },
    root: false, //é»˜è®¤ä¸ºtrue
    teamUUID: '',
  })

  if (response) {
    return response
  }
  return {
    body: {},
  }
